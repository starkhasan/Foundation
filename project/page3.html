Binary Search


		import java.util.Scanner;
		class demo
		{
    		public void binarysearch(int br[],int sr_ele)
    		{
        		int br_size=br.length;
        		int temp=0;

        		for(int i=0;i<br_size;i++)
        		{
            		for(int j=0;j<br_size-1;j++)
            		{
                		if(br[j]>br[j+1])
                		{
                    		temp=br[j];
                    		br[j]=br[j+1];
                    		br[j+1]=temp;
                		}
            		}
        		}

        		int mid=br_size/2;
        		if(sr_ele==br[mid])
        		{
            		System.out.println(sr_ele+" is found at "+mid);
            		System.exit(0);
        		}
        		else if(sr_ele<br[mid])
        		{
            		for(int i=0;i<mid;i++)
            		{
                		if(sr_ele==br[i])
                		{
                    		System.out.println(sr_ele+" is found at "+i);
                    		System.exit(0);
                		}
            		}
        		}
        		else
        		{
            		for(int i=mid+1;i<br_size;i++)
            		{
                		if(sr_ele==br[i])
                		{
                    		System.out.println(sr_ele+" is found at "+i);
                    		System.exit(0);
                		}
            		}
        		}

    		}
    		public static void main(String [] args)
    		{
        		demo br_search=new demo();
        		Scanner sc=new Scanner(System.in);
        		System.out.println("Enter Size :  ");
        		int ar_size=sc.nextInt();
        		System.out.println("Enter Element in Array : ");
        		int a[]=new int[ar_size];
        		for(int i=0;i<ar_size;i++)
        		{
            		a[i]=sc.nextInt();
        		}
        		System.out.println("Which Element Do you want to search : ");
        		int search_ele=sc.nextInt();
        		br_search.binarysearch(a,search_ele);

    		}
		}
